\section{Einteilung: 一段 oder 五段?}

\textbf{Beispiel:} 出る → $\boxtimes$ 一段\quad $\square$ 五段 \emph{(で = e vor dem る)}

Kreuz an. Denk an den Vokal vor dem る — und an die 例外.

\begin{multicols}{2}
\begin{enumerate}[itemsep=7pt]
\item 上げる \quad $\square$ 一段\quad $\square$ 五段
\item 立つ \quad $\square$ 一段\quad $\square$ 五段
\item \jterm{上}{のぼ}る \quad $\square$ 一段\quad $\square$ 五段
\item \jterm{止}{と}める \quad $\square$ 一段\quad $\square$ 五段
\item \jterm{引}{ひ}く \quad $\square$ 一段\quad $\square$ 五段
\item \jterm{広}{ひろ}げる \quad $\square$ 一段\quad $\square$ 五段
\item \jterm{入}{はい}る \quad $\square$ 一段\quad $\square$ 五段
\item \jterm{切}{き}る \quad $\square$ 一段\quad $\square$ 五段
\end{enumerate}
\end{multicols}

\section{Vergangenheit (過去形) — nur 一段}

\textbf{Beispiel:} 出る → 出\textbf{た}

\begin{enumerate}[itemsep=6pt]
\item 上げる → \lin[4cm]
\item 下げる → \lin[4cm]
\item \jterm{止}{と}める → \lin[4cm]
\item \jterm{広}{ひろ}げる → \lin[4cm]
\end{enumerate}

\section{Partikel}

\subsection*{4a. Lücke füllen — は · が · を}

\textbf{Beispiel:} 子 \lin[0.8cm] 火 \lin[0.8cm] 止めた。 → 子\textbf{が}火\textbf{を}止めた。

\begin{enumerate}[itemsep=8pt]
\item 犬 \lin[0.8cm] 出た。
\item 子 \lin[0.8cm] 手 \lin[0.8cm] 上げた。
\item 人 \lin[0.8cm] 木 \lin[0.8cm] \jterm{切}{き}る。
\item 子 \lin[0.8cm] 手 \lin[0.8cm] 下げた。
\end{enumerate}

\subsection*{4b. Was markiert welches Partikel?}

\textbf{が} und \textbf{を} hängen an festen Rollen des Verbs. \textbf{は} nicht — は
hebt nur hervor, \textbf{worum es geht}. Erste Zeile ist vorgemacht.

\begin{center}
\begin{tabular}{@{}lp{3cm}p{3.5cm}p{3.5cm}@{}}
\toprule
Verb & が & を & は \\
\midrule
出る & den Rausgehenden & nein — kein Objekt & worum es geht \\[10pt]
上げる & \lin[2.5cm] & \lin[3cm] & \lin[3cm] \\[10pt]
立つ & \lin[2.5cm] & \lin[3cm] & \lin[3cm] \\[10pt]
\jterm{止}{と}める & \lin[2.5cm] & \lin[3cm] & \lin[3cm] \\[10pt]
\jterm{引}{ひ}く & \lin[2.5cm] & \lin[3cm] & \lin[3cm] \\
\bottomrule
\end{tabular}
\end{center}

\subsection*{4c. を oder が? (transitiv / intransitiv)}

\textbf{Beispiel:} 手 \lin[0.8cm] 上げる (heben) → 手\textbf{を}上げる

\begin{enumerate}[itemsep=7pt]
\item 手 \lin[0.8cm] \jterm{上}{あ}がる (sich heben) → \lin[3cm]
\item 火 \lin[0.8cm] 止める (etwas anhalten) → \lin[3cm]
\item 火 \lin[0.8cm] \jterm{止}{と}まる (von selbst stehenbleiben) → \lin[3cm]
\item 手 \lin[0.8cm] 下げる (senken) → \lin[3cm]
\end{enumerate}

\subsection*{4d. が oder は? (die stille Frage entscheidet)}

\textbf{Beispiel:} [Antwort auf „誰が出た？"] 犬 \lin[0.8cm] 出た。 → 犬\textbf{が}出た。

\begin{enumerate}[itemsep=8pt]
\item {}[Antwort auf „\jterm{誰}{だれ}が立つ？"] 子 \lin[0.8cm] 立つ。
\item {}[Es geht um den Hund — was ist mit ihm?] 犬 \lin[0.8cm] 出た。
\item {}[Antwort auf „wer hob die Hand?"] 子 \lin[0.8cm] 手を上げた。
\item {}[Thema: das Kind] 子 \lin[0.8cm] 手を下げる。
\end{enumerate}

\subsection*{4e. Fehler finden}

\textbf{Beispiel:} 犬\textbf{を}出る。 → 出る hat kein Objekt → 犬\textbf{が}出る。

\begin{enumerate}[itemsep=8pt]
\item 手\textbf{が}上げる。 → \lin[6cm]
\item 子\textbf{を}立つ。 → \lin[6cm]
\item 犬\textbf{を}出た。 → \lin[6cm]
\end{enumerate}
